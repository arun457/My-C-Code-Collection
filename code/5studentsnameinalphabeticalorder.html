//WAP to read name, rollno and marks of 5 different students as
//the three members of a structure name 'student'. Display the name and corresponding rollno and
// marks of the students sorted in an alphabetical order using
//user define function and display sorted data from main function. [10]
#include <stdio.h>
#include <conio.h>
#include <stdlib.h>
#include <string.h>
#define loopsize 2

struct student
{
    char name[50];
    int rollno;
    float marks;
};

struct student sorting(struct student sorting[loopsize]);

void main()
{
    int i, j;
    struct student s[loopsize], temp;
    system("cls");

    printf("Enter the name rollno and marks of 5 students : \n\n");

    for (i = 0; i < loopsize; i++)
    {
        printf("Enter the credentials of %dth students : \n", i + 1);
        printf("Enter name: ");
        scanf("%s", &s[i].name);
        printf("Enter rollno: ");
        scanf("%d", &s[i].rollno);
        printf("Enter marks: ");
        scanf("%f", &s[i].marks);
        printf("\n");
    }

    sorting(s);

    printf("Name\tRollno\tMarks\n");
    for (i = 0; i < loopsize; i++)
    {
        printf("%s\t%d\t%f", s[i].name, s[i].rollno, s[i].marks);
        printf("\n");
    }
}
struct student sorting(struct student sorting[loopsize])
{
    int i, j;

    struct student temp;

    for (i = 0; i < loopsize; i++)
    {
        for (j = 0; j < loopsize; j++)
        {
            if ((strcmp(sorting[i].name, sorting[j].name)) < 0)
            {
                temp = sorting[i];
                sorting[i] = sorting[j];
                sorting[j] = temp;
            }
        }
    }
}
