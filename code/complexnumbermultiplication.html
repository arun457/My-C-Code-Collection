//WAP to represent complex number by a structure with real and imaginary as members.
//Take 2 complex number as input from user into sturcture variable and display a third complex number
//which is the multiplication of the entered complex numbers.
// Use user defined function to perform multiplication and display the result from another function display(). [10]
#include <stdio.h>
#include <conio.h>
#include <stdlib.h>

struct complexnumber
{
    int real;
    int img;
};

struct complexnumber multiplication(struct complexnumber c1, struct complexnumber c2);
struct complexnumber display(struct complexnumber display);

void main()
{
    struct complexnumber ca, cb, result;
    system("cls");
    printf("Enter the First real numer : ");
    scanf("%d%d", &ca.real, &ca.img);
    printf("Enter the First real numer : ");
    scanf("%d%d", &cb.real, &cb.img);
    result = multiplication(ca, cb);

    display(result);
}

struct complexnumber multiplication(struct complexnumber c1, struct complexnumber c2)
{
    struct complexnumber mul;
    mul.real = c1.real * c2.real - c1.img * c2.img;
    mul.img = c1.real * c2.img + c1.img * c2.real;
    return (mul);
}
struct complexnumber display(struct complexnumber display)
{
    printf("The multiplication is : %d+%di", display.real, display.img);
}
